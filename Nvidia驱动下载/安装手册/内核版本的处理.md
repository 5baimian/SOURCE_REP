# 处理多个内核
```
//查看已经安装内核
dpkg --get-selections |grep linux-image

//默认内核版本
linux-image-6.8.0-40-generic			install

//查看启动顺序
grep menuentry /boot/grub/grub.cfg

//修改内核启动顺序
sudo gedit /etc/default/grub

打开/etc/default/grub文件
将GRUB_DEFAULT=0注释，新增
GRUB_DEFAULT=“Advanced options for Ubuntu>Ubuntu, with Linux 5.4.0-91-generic”//版本要更改



```
